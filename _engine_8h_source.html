276  template<typename T>
277  ComponentPool<T> *getOrCreateComponentPool() {
278  auto type = getComponentType<T>();
279  if (type >= componentPoolsByType.size())
280  componentPoolsByType.resize(type + 1);
281  auto *pool = (ComponentPool<T> *)componentPoolsByType[type];
282  if (!pool) {
283  pool = new ComponentPool<T>(componentPoolInitialSize, componentPoolMaxSize);
284  componentPoolsByType[type] = pool;
285  }
286  return pool;
287  }
